# P2PK(Pay to Public Key)
* input script:
    * PUSHDATA(Sig)
* output script:
    * PUSHDATA(PubKey)
    * CHECKSIG
# P2PKH(Pay to Public Key Hash)
* input script:
    * PUSHDATA(Sig)
    * PUSHDATA(PubKey)
* output script:
    * DUP
    * HASH160
    * PUSHDATA(PubKeyHash)
    * EQUALVERIFY
    * CHECKSIG
# P2SH(Pay to Script Hash)
* input script:
    * ...
    * PUSHDATA(Sig)
    * ...
    * PUSHDATA(serialized redeemScript)
* output script:
    * HASH160
    * PUSHDATA(redeemScriptHash)
    * EQUAL
